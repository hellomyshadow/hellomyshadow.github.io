        public ResponseEntity<ExceptionResult> handleLyExt(LyException ext) {
            return ResponseEntity.status(ext.getExceptionEnum().getCode())
                    .body(new ExceptionResult(ext.getExceptionEnum()));
        }
        @ExceptionHandler(RuntimeException.class)
        public ResponseEntity<String> handleRuntimeExt(RuntimeException ext) {
            return ResponseEntity.status(HttpStatus.BAD_REQUEST).body(ext.getMessage());
        }
    }

5. 分页助手PageHelper和通用Mapper是同一个作者，但PageHelper不依赖于通用Mapper，可以单独使用;
    1. 在工具工程ly-common中，创建响应的分页对象：com.leyou.common.vo.PageResult
    @Data
    public class PageResult<T> {
        private Long total;   //总条数
        private Integer pages;   //总页数
        private List<T> items;   //当前页的数据
        public PageResult() {}
        public PageResult(Long total, List<T> items) {
            this.total = total;
            this.items = items;
        }
        public PageResult(Long total, Integer pages, List<T> items) {
            this.total = total;
            this.pages = pages;
            this.items = items;
        }
    }
    2. 在调用查询语句之前，使用PageHelper实现分页查询
    public PageResult<Brand> findByPage(Integer, page, Integer rows) {
        //分页：使用拦截器，在SQL语句后面拼接limit
        PageHelper.startPage(page, rows);
        //查询
        List<Brand> list = mapper.selectAll();
        //解析查询结果
        PageInfo<Brand> info = new PageInfo<>(list);
        return new PageResult<>(info.getTotal(), info.getPages(), list);
    }
